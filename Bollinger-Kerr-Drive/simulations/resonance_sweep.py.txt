import numpy as np
import pandas as pd  # For nicer table output, but optional

# Target frequency and tolerance
target_f = 10.3e6  # 10.3 MHz in Hz
tolerance = 0.1    # 10% tolerance

# Parameter ranges
u2_min = 1e4       # Start from 10^4 m/s to focus on plausible range
u2_max = 1e8       # Up to ~c
L_min = 1e-3       # 1 mm
L_max = 1e0        # 1 m

# Number of points for logspace sweep
num_points = 50    # Reduced for quick sim

# Generate log-spaced values
u2_vals = np.logspace(np.log10(u2_min), np.log10(u2_max), num_points)
L_vals = np.logspace(np.log10(L_min), np.log10(L_max), num_points)

# List to hold matching pairs
matches = []

# Sweep and find matches
for u2 in u2_vals:
    for L in L_vals:
        f = u2 / (2 * L)  # Fundamental frequency f = u2 / (2L)
        if abs(f - target_f) / target_f <= tolerance:
            matches.append((u2, L, f))

# Sort by f for nicer output
matches.sort(key=lambda x: x[2])

# Limit to 10 examples
matches = matches[:10]

# Print table
if matches:
    df = pd.DataFrame(matches, columns=['u₂ (m/s)', 'L (m)', 'f (Hz)'])
    # Format for scientific notation
    pd.set_option('display.float_format', '{:.2e}'.format)
    print(df.to_markdown(index=False))
else:
    print("No matches found within tolerance.")
