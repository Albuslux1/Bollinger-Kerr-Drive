def analyze_he4_extremes():
    """Analyze He-4 under extreme conditions"""
    
    print("He-4 UNDER EXTREME CONDITIONS:")
    print("=" * 50)
    
    # Theoretical maximum second sound in He-4
    # From Landau's two-fluid model: u₂² = (ρ_s/ρ_n) * (∂P/∂ρ)_s * (T s² / c_p)
    
    # Extreme scenario calculations
    scenarios = [
        ("Standard He-II", 20, 0.1, 2.17),
        ("High Pressure (100 bar)", 50, 10, 1.5), 
        ("Near Absolute Zero", 100, 0.01, 0.1),
        ("Theoretical Max (Landau limit)", 300, 1000, 0.001),
        ("Hypothetical Supersolid", 1000, 10000, 0.0001)
    ]
    
    print("\nSecond Sound Velocity in Extreme He-4 Scenarios:")
    print("Scenario                u₂ (m/s)  P (bar)   T (K)")
    print("-" * 50)
    
    for scenario, u2, pressure_bar, temp in scenarios:
        print(f"{scenario:25} {u2:6.0f}     {pressure_bar:6.0f}    {temp:6.3f}")
    
    return scenarios

def plot_he4_vs_requirements():
    """Plot He-4 capabilities vs 10.3 MHz requirements"""
    
    L_range = np.logspace(-3, 0, 100)  # 1 mm to 1 m
    u2_required = 2 * L_range * target_f
    
    # He-4 scenarios
    scenarios = [
        ("Standard He-II", 20),
        ("High Pressure", 50), 
        ("Near 0K", 100),
        ("Landau Limit", 300),
        ("Supersolid", 1000)
    ]
    
    plt.style.use('dark_background')
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 5))
    
    # Velocity comparison
    ax1.loglog(L_range * 100, u2_required, 'c-', linewidth=3, 
               label='Required for 10.3 MHz')
    
    for scenario, u2 in scenarios:
        ax1.axhline(u2, linestyle='--', label=f'{scenario} (u₂ = {u2} m/s)')
    
    ax1.axhline(c, color='w', linestyle=':', alpha=0.7, label='Speed of Light')
    
    ax1.set_xlabel('Cavity Length L (cm)')
    ax1.set_ylabel('Second Sound Velocity u₂ (m/s)')
    ax1.set_title('He-4 Capabilities vs 10.3 MHz Requirements', fontsize=12)
    ax1.legend(fontsize=8)
    ax1.grid(True, alpha=0.3)
    
    # Maximum achievable frequency
    max_frequencies = []
    scenario_names = []
    
    for scenario, u2 in scenarios:
        # For 1 cm cavity
        f_max = u2 / (2 * 0.01)  # 1 cm cavity
        max_frequencies.append(f_max / 1e3)  # in kHz
        scenario_names.append(scenario)
    
    ax2.bar(scenario_names, max_frequencies, color=['red', 'orange', 'yellow', 'green', 'cyan'])
    ax2.axhline(target_f / 1e3, color='white', linestyle='--', linewidth=2,
               label='Target 10.3 MHz')
    ax2.set_ylabel('Maximum Frequency (kHz)')
    ax2.set_title('Max Frequency in 1 cm Cavity with He-4')
    ax2.tick_params(axis='x', rotation=45)
    ax2.legend()
    ax2.grid(True, alpha=0.3)
    
    plt.tight_layout()
    plt.savefig('he4_extreme_analysis.png', dpi=150, facecolor='black')
    plt.show()
    
    return max_frequencies
